Template filtering practice:

{{Dict | join:" ; "}}

{{Datetime | date:"SHORT_DATE_FORMAT"}} -> {{Datetime | date}}

{{Datetime | date:"D d M Y"}}, {{Datetime | time:"h:i A"}}

{{"String" | default:"Blah Blah"}} {{"" | default:"Was Empty"}}

{{3 | add:"2"}} {{"is" | add:" added "}}

{{"C isc" | cut:"c"}} {{"c u t " | cut:" "}}

{{"THIS WAS UPPERCASE" | lower}} & {{"this was lowercase" | upper}}

{{"only FIRST letter capitalized" | capfirst}} & {{"all words start with capital letter" | title}}

My age: {{NewDate | timesince}} & {{OldDate | timesince:NewDate}}

Length of " " is {{" " | length}}, not {{"" | length}}

{{"This line is ok" | truncatechars:15}}, {{"This line is too long" | truncatechars:20}}

{{"Just enough words" | truncatewords:3}} ({{"Just enough words" | wordcount}}). {{"Or is it too many words?" | truncatewords:5}}
